import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import palette from './theme/palette';

<Meta title="Colors" />

# Palette

<ColorPalette>
  {['red', 'blue', 'green', 'purple', 'yellow', 'grey', 'basic'].map(
    (colorName) => (
      <ColorItem
        title={colorName[0].toUpperCase() + colorName.slice(1)}
        colors={palette.swatches[colorName]}
      />
    )
  )}
</ColorPalette>

# Fun colors

<ColorPalette>
  {Object.entries(palette.swatches)
    .filter((tuple) => 'dark' in tuple[1])
    .map(([colorName, colorGroup]) => (
      <ColorItem
        title={colorName[0].toUpperCase() + colorName.slice(1)}
        colors={{
          dark: colorGroup.dark.color,
          light: colorGroup.light.color,
          medium: colorGroup.medium.color,
        }}
      />
    ))}
</ColorPalette>
